# FizzBuzz - Print on Multiples of 3 and/or 5

```swift
for i in 1...100 {
    let multipleOf3 = (i % 3 == 0)
    let multipleOf5 = (i % 5 == 0)
    if multipleOf3 && multipleOf5 {
        print("FizzBuzz")
    } else if multipleOf3 {
        print("Fizz")
    } else if multipleOf5 {
        print("Buzz")
    } else {
        print(i)
    }
}
```
