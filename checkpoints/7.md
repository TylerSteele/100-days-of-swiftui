# Animal Hierarchy

```swift
class Animal {
    let numberOfLegs: Int
    func speak() {

    }
    init(numberOfLegs: Int = 4) {
        self.numberOfLegs = numberOfLegs
    }
}

class Dog: Animal {
    override func speak() {
        print("Woof, woof!")
    }
    init() {
        super.init(numberOfLegs: 4)
    }
}

class Corgi: Dog {

}

class Poodle: Dog {

}

class Cat: Animal {
    override func speak() {
        print("Puuuuuuurrr.")
    }
    let isTame: Bool

    init(isTame: Bool) {
        self.isTame = isTame
        super.init(numberOfLegs: 4)
    }
}

class Persian: Cat {
    init() {
        super.init(isTame: true)
    }
}

class Lion: Cat {
    init() {
        super.init(isTame: false)
    }
}
```
