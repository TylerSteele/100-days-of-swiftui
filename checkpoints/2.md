# Count Array Strings and Remove Duplicates

```swift
let names = ["Tyler", "Dylan", "Archie", "Magdalane", "Dylan", "Magdalane"]

let numberOfNames = names.count             // 6
let uniqueNames = Set(names)
let numberOfUniqueNames = uniqueNames.count // 4

print("List of names: \(names)")
print("Number of names: \(numberOfNames)")
print("List of unique names: \(uniqueNames)")
print("Number of unique names: \(numberOfUniqueNames)")
```
